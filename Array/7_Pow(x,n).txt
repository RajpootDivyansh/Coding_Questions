Implement pow(x, n), which calculates x raised to the power n (i.e., xn).

class Solution {
public:
    double myPow(double x, int n) {
        double ans = 1;
        long long z = n;  

        if (z < 0) {
            x = 1 / x;
            z = -z;
        }

        while (z > 0) {
            if (z % 2 == 1) {
                ans = ans * x;
            }
            x = x * x;
            z = z / 2;
        }

        return ans;
    }
};
